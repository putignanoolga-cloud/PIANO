function getMondayOf(offset=0){
  const today=new Date();
  const dow=today.getDay();
  const monday=new Date(today);
  monday.setDate(today.getDate()-((dow+6)%7)+offset*7);
  monday.setHours(0,0,0,0);
  return monday;
}
function getWeekDays(offset=0){
  const monday=getMondayOf(offset);
  const today=new Date(); today.setHours(0,0,0,0);
  return Array.from({length:7},(_,i)=>{
    const d=new Date(monday);
    d.setDate(monday.getDate()+i);
    return{key:`${d.getFullYear()}-${d.getMonth()}-${d.getDate()}`,dayIndex:i,label:DAY_NAMES[i],num:d.getDate(),month:d.toLocaleDateString('it-IT',{month:'short'}),isToday:d.getTime()===today.getTime(),dateObj:d};
  });
}
function getWeekLabel(offset=0){
  const days=getWeekDays(offset);
  const fmt=d=>d.toLocaleDateString('it-IT',{day:'numeric',month:'short'});
  if(offset===0)return'Questa settimana';
  if(offset===1)return'Prossima settimana';
  if(offset===-1)return'Settimana scorsa';
  return`${fmt(days[0].dateObj)} – ${fmt(days[6].dateObj)}`;
}

      <div className="header">
        <div style={{display:'flex',alignItems:'center',justifyContent:'space-between'}}>
          <span className="app-name">piano.</span>
          <span style={{fontSize:11,fontWeight:600,padding:'2px 8px',borderRadius:10,background:remaining<=3?'#FFF0F0':'#F0ECFC',color:remaining<=3?'#FF6B6B':'#9B8EC4'}}>🔍 {remaining}/{MAX_SEARCHES}</span>
        </div>
        <div className="week-nav">
          <button className="nav-btn" onClick={()=>setWeekOffset(w=>w-1)}>‹</button>
          <button className="today-btn" onClick={()=>setWeekOffset(0)}>Oggi</button>
          <span className="week-label-text">{getWeekLabel(weekOffset)}</span>
          <button className="nav-btn" onClick={()=>setWeekOffset(w=>w+1)}>›</button>
        </div>
      </div>
      <div className="search-bar">
        <div className="search-row">
          <input className="search-input" placeholder="Aggiungi una priorità..." value={text} onChange={e=>setText(e.target.value)} onKeyDown={e=>e.key==='Enter'&&add()}/>
          <button className="search-btn" onClick={add}>+</button>
        </div>
      </div>
      <div className="tabs">
        <button className={`tab${tab==='priorita'?' active':''}`} onClick={()=>setTab('priorita')}>Priorità</button>
        <button className={`tab${tab==='settimana'?' active':''}`} onClick={()=>setTab('settimana')}>Settimana</button>
      </div>
      <div className="content">
        {tab==='priorita'?(
          <div className="prio-list">
            {unassigned.length>0&&<><div className="sec-label">Non assegnate ({unassigned.length})</div>{unassigned.map(p=>renderPcard(p))}</>}
            {DAYS.map(d=>{const dp=items.filter(p=>p.dayKey===d.key);if(!dp.length)return null;const c=DAY_COLORS[d.dayIndex];return(<div key={d.key}><div className="sec-label" style={{color:c.text}}>{d.label} {d.num} {d.month}</div>{dp.map(p=>renderPcard(p))}</div>);})}
            {items.length===0&&<div style={{textAlign:'center',padding:'40px 0',color:'#C8BFB0',fontSize:14}}>Aggiungi la tua prima priorità 👆</div>}
          </div>
        ):(
          <div className="week-view">
            <div className="unassigned">
              <span className="u-label">Libere</span>
              {unassigned.length===0?<span style={{fontSize:12,color:'#C8BFB0',fontStyle:'italic'}}>Tutte assegnate 🎉</span>:unassigned.map(p=>{const c=PRIO_COLORS[p.ci%PRIO_COLORS.length];return(<div key={p.id} className="mini" style={{background:c.light,color:c.text}} onClick={()=>setModal(p.id)}><div className="mini-dot" style={{background:c.bg}}/>{p.text}</div>);})}
            </div>
            <div className="days-wrap">
              {DAYS.map(d=>{
                const dp=items.filter(p=>p.dayKey===d.key);
                const isOpen=expandedDays[d.key];
                const c=DAY_COLORS[d.dayIndex];
                return(
                  <div key={d.key} className="day-row" style={{background:c.light}}>
                    <div className="day-row-header" onClick={()=>toggleDayExpand(d.key)}>
                      <div className="day-pill" style={{background:d.isToday?'#1C1C1C':c.bg}}>
                        <span className="dn" style={{color:'rgba(255,255,255,.7)'}}>{d.label}</span>
                        <span className="dd" style={{color:'white'}}>{d.num}</span>
                      </div>
                      <div className="day-tasks-preview">
                        {dp.length>0?dp.slice(0,2).map(p=><span key={p.id} className="day-chip" style={{background:'rgba(255,255,255,.8)',color:c.text}}>{p.text}</span>):<span style={{fontSize:12,color:c.text,opacity:.5,fontStyle:'italic'}}>Nessuna priorità</span>}
                        {dp.length>2&&<span style={{fontSize:11,color:c.text,opacity:.6}}>+{dp.length-2}</span>}
                      </div>
                      <span style={{fontSize:12,color:c.text,opacity:.5,marginLeft:'auto',flexShrink:0}}>{isOpen?'▲':'▼'}</span>
                    </div>
                    {isOpen&&dp.length>0&&(
                      <div className="day-tasks-expanded">
                        {dp.map(p=>(
                          <div key={p.id} className="day-task-row">
                            <div className="day-task-dot" style={{background:c.bg}}/>
                            <span className="day-task-text">{p.text}</span>
                            {p.subtasks.length>0&&<span style={{fontSize:10,color:c.text,background:'rgba(255,255,255,.7)',padding:'2px 6px',borderRadius:4,flexShrink:0}}>{p.subtasks.filter(s=>s.done).length}/{p.subtasks.length}</span>}
                            <button className="day-task-move" onClick={()=>setModal(p.id)}>Sposta</button>
                          </div>
                        ))}
                      </div>
                    )}
                  </div>
                );
              })}
            </div>
          </div>
        )}
      </div>
      {modal&&modalItem&&(
        <div className="overlay" onClick={()=>setModal(null)}>
          <div className="sheet" onClick={e=>e.stopPropagation()}>
            <div className="sheet-title">Quando?</div>
            <div className="sheet-sub"><div style={{width:8,height:8,borderRadius:'50%',background:getPColor(modalItem).bg,flexShrink:0}}/>{modalItem.text}</div>
            <div className="day-grid">
              {DAYS.map(d=>(<button key={d.key} className={`day-opt${modalItem.dayKey===d.key?' sel':''}`} onClick={()=>{assign(modalItem.id,d.key);setModal(null);}}><div className="don">{d.label}</div><div className="dod">{d.num}</div></button>))}
              <button className="unassign-opt" onClick={()=>{assign(modalItem.id,null);setModal(null);}}><div className="don" style={{color:'#9E9488'}}>Rimuovi</div><div className="dod" style={{color:'#9E9488'}}>—</div></button>
            </div>
            <button className="cancel" onClick={()=>setModal(null)}>Annulla</button>
          </div>
        </div>
      )}
    </div>
  );
}
ReactDOM.createRoot(document.getElementById('root')).render(<App/>);
